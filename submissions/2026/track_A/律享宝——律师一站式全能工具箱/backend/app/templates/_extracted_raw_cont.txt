@RestController
@RequestMapping("/api/templates")
public class TemplateController {

    @Autowired
    private Configuration freemarkerConfig;
    
    @Autowired
    private TemplateService templateService;

    @PostMapping("/{id}/generate")
    public ResponseEntity<byte[]> generateDocument(
            @PathVariable String id,
            @RequestBody Map<String, Object> formData) throws Exception {
        
        // 获取模板内容
        Template template = templateService.getTemplateById(id);
        String templateContent = template.getTemplateContent();
        
        // 渲染模板
        StringWriter writer = new StringWriter();
        freemarkerConfig.getTemplate(templateContent).process(formData, writer);
        String htmlContent = writer.toString();
        
        // 转换为PDF或Word
        byte[] documentBytes = DocumentConverter.convertToPdf(htmlContent);
        
        // 返回文件
        HttpHeaders headers = new HttpHeaders();
        headers.setContentType(MediaType.APPLICATION_PDF);
        headers.setContentDispositionFormData("attachment", template.getTemplateName() + ".pdf");
        
        return new ResponseEntity<>(documentBytes, headers, HttpStatus.OK);
    }
}
